// Online C++ compiler to run C++ program online
#include <iostream>
#include <stack>
using namespace std;

int main() 
{
    stack <int> s;
    s.push(2);
    s.push(9);
    cout<<"printing top element : "<<s.top()<<endl;
    cout<<"empty or not  : "<<s.empty()<<endl;
    cout<<"size : "<<s.size()<<endl;


    return 0;
}
